813 static int glad_glx_has_extension(Display *display, int screen, const char *ext) {
814 #ifndef GLX_VERSION_1_1
815  (void) display;
816  (void) screen;
817  (void) ext;
818 #else
819  const char *terminator;
820  const char *loc;
821  const char *extensions;
822 
823  if (glXQueryExtensionsString == NULL) {
824  return 0;
825  }
826 
827  extensions = glXQueryExtensionsString(display, screen);
828 
829  if(extensions == NULL || ext == NULL) {
830  return 0;
831  }
832 
833  while(1) {
834  loc = strstr(extensions, ext);
835  if(loc == NULL)
836  break;
837 
838  terminator = loc + strlen(ext);
839  if((loc == extensions || *(loc - 1) == ' ') &&
840  (*terminator == ' ' || *terminator == '\0')) {
841  return 1;
842  }
843  extensions = terminator;
844  }
845 #endif
846 
847  return 0;
848 }
